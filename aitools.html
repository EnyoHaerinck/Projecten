import pygame import sys # Initialize Pygame pygame.init() # Screen dimensions SCREEN_WIDTH = 800 SCREEN_HEIGHT = 600 # Colors WHITE = (255, 255, 255) BLACK = (0, 0, 0) # Paddle dimensions PADDLE_WIDTH = 10 PADDLE_HEIGHT = 100 # Ball dimensions BALL_SIZE = 10 # Speeds PADDLE_SPEED = 5 BALL_SPEED_X = 7 BALL_SPEED_Y = 7 # Initialize screen screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("Pong") # Clock for controlling the frame rate clock = pygame.time.Clock() # Paddle positions left_paddle = pygame.Rect(20, (SCREEN_HEIGHT - PADDLE_HEIGHT) // 2, PADDLE_WIDTH, PADDLE_HEIGHT) right_paddle = pygame.Rect(SCREEN_WIDTH - 30, (SCREEN_HEIGHT - PADDLE_HEIGHT) // 2, PADDLE_WIDTH, PADDLE_HEIGHT) # Ball position and velocity ball = pygame.Rect(SCREEN_WIDTH // 2, SCREEN_HEIGHT // 2, BALL_SIZE, BALL_SIZE) ball_velocity = [BALL_SPEED_X, BALL_SPEED_Y] # Scores left_score = 0 right_score = 0 # Font for displaying scores font = pygame.font.Font(None, 74) button_font = pygame.font.Font(None, 50) # Win condition WIN_SCORE = 10 def reset_ball(): ball.center = (SCREEN_WIDTH // 2, SCREEN_HEIGHT // 2) ball_velocity[0] *= -1 def draw_objects(): screen.fill(BLACK) pygame.draw.rect(screen, WHITE, left_paddle) pygame.draw.rect(screen, WHITE, right_paddle) pygame.draw.ellipse(screen, WHITE, ball) pygame.draw.aaline(screen, WHITE, (SCREEN_WIDTH // 2, 0), (SCREEN_WIDTH // 2, SCREEN_HEIGHT)) left_text = font.render(str(left_score), True, WHITE) right_text = font.render(str(right_score), True, WHITE) screen.blit(left_text, (SCREEN_WIDTH // 4, 20)) screen.blit(right_text, (SCREEN_WIDTH * 3 // 4, 20)) def draw_win_screen(winner): screen.fill(BLACK) win_text = font.render(f"{winner} wint!", True, WHITE) restart_text = button_font.render("Opnieuw spelen", True, WHITE) win_rect = win_text.get_rect(center=(SCREEN_WIDTH // 2, SCREEN_HEIGHT // 3)) button_rect = restart_text.get_rect(center=(SCREEN_WIDTH // 2, SCREEN_HEIGHT // 2)) screen.blit(win_text, win_rect) pygame.draw.rect(screen, WHITE, button_rect, 2) screen.blit(restart_text, button_rect) return button_rect def reset_game(): global left_score, right_score, ball_velocity left_score = 0 right_score = 0 reset_ball() # Main game loop running = True win_screen = False winner = "" while running: for event in pygame.event.get(): if event.type == pygame.QUIT: running = False if win_screen and event.type == pygame.MOUSEBUTTONDOWN: if button_rect.collidepoint(event.pos): reset_game() win_screen = False if not win_screen: # Get keys keys = pygame.key.get_pressed() if keys[pygame.K_w] and left_paddle.top > 0: left_paddle.y -= PADDLE_SPEED if keys[pygame.K_s] and left_paddle.bottom < SCREEN_HEIGHT: left_paddle.y += PADDLE_SPEED if keys[pygame.K_UP] and right_paddle.top > 0: right_paddle.y -= PADDLE_SPEED if keys[pygame.K_DOWN] and right_paddle.bottom < SCREEN_HEIGHT: right_paddle.y += PADDLE_SPEED # AI control for left paddle if ball.centery < left_paddle.centery and left_paddle.top > 0: left_paddle.y -= PADDLE_SPEED if ball.centery > left_paddle.centery and left_paddle.bottom < SCREEN_HEIGHT: left_paddle.y += PADDLE_SPEED # Move ball ball.x += ball_velocity[0] ball.y += ball_velocity[1] # Ball collision with top and bottom if ball.top <= 0 or ball.bottom >= SCREEN_HEIGHT: ball_velocity[1] *= -1 # Ball collision with paddles if ball.colliderect(left_paddle) or ball.colliderect(right_paddle): ball_velocity[0] *= -1 # Ball out of bounds if ball.left <= 0: right_score += 1 reset_ball() if ball.right >= SCREEN_WIDTH: left_score += 1 reset_ball() # Check win condition if left_score >= WIN_SCORE: win_screen = True winner = "Links" elif right_score >= WIN_SCORE: win_screen = True winner = "Rechts" # Draw everything draw_objects() else: button_rect = draw_win_screen(winner) # Update display pygame.display.flip() # Cap the frame rate clock.tick(60) pygame.quit() sys.exit()
